func sqrt(x float64, p float64) float64 {
	l, r := 0.0, x
	for {
		mid := l + (r-l)/2.0
		t := mid * mid
		if math.Abs(x-t) < p {
			return mid
		} else if t < x {
			l = mid + 1
		} else {
			r = mid - 1
		}
	}
}

func lengthOfLIS(nums []int) int {
	ret := make([]int, 0, len(nums))
	for i := 0; i < len(nums); i++ {
		t := find(ret, nums[i])
		if t == -1 {
			ret = append(ret, nums[i])
		} else {
			ret[t] = nums[i]
		}
	}
	return len(ret)
}

func find(nums []int, k int) int {
	ll := len(nums) - 1
	if ll < 0 {
		return -1
	}
	l, r := 0, ll
	for l < r {
		mid := l + (r-l)>>1
		if nums[mid] == k {
			return mid
		} else if nums[mid] > k {
			r = mid
		} else {
			l = mid + 1
		}
	}
	if l == ll && nums[l] < k {
		return -1
	}
	return l
}
